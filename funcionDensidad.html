import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from scipy.stats import gamma

# Supongamos que estos son los tiempos de atención originales (en minutos)
# También ajustaremos la distribución gamma a estos tiempos
tiempos_atencion_originales = np.array([35, 45, 10, 25, 50, 22])

# Parámetros de la distribución gamma ajustada para los tiempos de atención


# Parámetros de la distribución gamma ajustada
shape_atencion, loc_atencion, scale_atencion = gamma.fit(df_filtered['Tiempo_Atencion'], floc=0)

# Supongamos que estos son los tiempos de viaje originales (en minutos)
tiempos_viaje_originales = np.array([50, 45, 60, 30, 45, 35])

# Definir los porcentajes de variación para los tiempos de viaje
variacion_inferior = 0.9  # 10% menos
variacion_superior = 1.3  # hasta 30% más

# Número de simulaciones
num_simulaciones = 1000

# Crear una matriz para almacenar los tiempos simulados totales
tiempos_totales_simulados = []

# Hacer la simulación para cada ruta
for _ in range(num_simulaciones):
    tiempo_total = 0  # Tiempo total inicializado en 0
    
    # Iterar sobre cada cliente y viaje
    for i in range(len(tiempos_atencion_originales)):
        # Simular tiempo de atención usando la distribución gamma
        tiempo_atencion_simulado = gamma.rvs(shape_atencion, loc=loc_atencion, scale=scale_atencion)
        
        # Agregar el tiempo de atención al total
        tiempo_total += tiempo_atencion_simulado
        
        # Simular el tiempo de viaje con la variación definida (solo si no es el último cliente)
        if i < len(tiempos_viaje_originales):
            tiempo_viaje_simulado = np.random.uniform(low=tiempos_viaje_originales[i] * variacion_inferior, high=tiempos_viaje_originales[i] * variacion_superior)
            # Agregar el tiempo de viaje al total
            tiempo_total += tiempo_viaje_simulado
    
    # Agregar el tiempo total de esta simulación a la lista
    tiempos_totales_simulados.append(tiempo_total)

# Convertir los tiempos simulados totales a un DataFrame para análisis
df_simulaciones = pd.DataFrame({'Tiempo_Total': tiempos_totales_simulados})

# Mostrar los primeros resultados de la simulación
df_simulaciones.head()

# Graficar la distribución de los tiempos totales simulados
sns.histplot(df_simulaciones['Tiempo_Total'], kde=True, color='blue')
plt.title('Distribución de Tiempos Totales Simulados (Atención + Viajes)')
plt.xlabel('Tiempo Total de Viaje y Atención (minutos)')
plt.ylabel('Frecuencia')
plt.show()

# Mostrar los estadísticos descriptivos de los tiempos totales simulados
df_simulaciones['Tiempo_Total'].describe()
